  <h2>🎤 Bilingual Voice Translator (Sarvam)</h2>

  <select id="role">

    <option value="teacher">Teacher (Hindi to Punjabi)</option>

    <option value="student">Student (Punjabi to Hindi)</option>

  </select>

  <button onclick="startRecording()">🎙️ Record & Translate</button>


  <div class="log" id="result">

    <b>Transcript:</b> <span id="transcript">-</span><br>

    <b>Translated:</b> <span id="translated">-</span>

  </div>


  <script>

    async function startRecording() {

      const btn = document.querySelector("button");

      btn.disabled = true;

      btn.innerText = "Recording...";


      const stream = await navigator.mediaDevices.getUserMedia({ audio: true });

      const mediaRecorder = new MediaRecorder(stream);

      let chunks = [];


      mediaRecorder.ondataavailable = e => chunks.push(e.data);

      mediaRecorder.onstop = async () => {

        const blob = new Blob(chunks, { type: 'audio/wav' });

        const reader = new FileReader();

        reader.onloadend = async () => {

          const b64 = reader.result;

          const role = document.getElementById("role").value;

          const res = await fetch("/translate_play", {

            method: "POST",

            headers: { "Content-Type": "application/json" },

            body: JSON.stringify({ audio: b64, role })

          });

          const data = await res.json();

          document.getElementById("transcript").innerText = data.original || "-";

          document.getElementById("translated").innerText = data.translated || "-";

          btn.innerText = "🎙️ Record & Translate";

          btn.disabled = false;

        };

        reader.readAsDataURL(blob);

      };


      mediaRecorder.start();

      setTimeout(() => {

        mediaRecorder.stop();

        stream.getTracks().forEach(t => t.stop());

      }, 5000);

    }

  </script>

</body>
